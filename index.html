Ghost Lib UI Documentation

Le moteur UI le plus rapide et complet pour Luau. Supporte les icônes conditionnelles.

Bootload

local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/gigatchadeed-design/GhostLIB-luau/main/GhostLibUi.lua"))()

Notifications NEW

Affichez des alertes n'importe où sans avoir créé de fenêtre.

Library:Notify({
    Title = "Ghost Lib Loaded",
    Content = "Welcome, " .. game.Players.LocalPlayer.Name,
    Duration = 5,
    Icon = "rbxassetid://7733658504" -- Alert-triangle ID
})

Window & Tabs

CreateWindow (title, icon) 
local Window = Library:CreateWindow("Ghost Hub", "rbxassetid://7743868000")
CreateTab (name, icon) 
local MainTab = Window:CreateTab("Combat", "rbxassetid://7733799901")
MainTab.Show() -- Affiche l'onglet par défaut

Toggles

CreateToggle (name, callback, icon) 
MainTab:CreateToggle("Auto Kill", function(state)
    print("Toggled: ", state)
end, "rbxassetid://7733674079") -- Axe Icon

Sliders

CreateSlider (name, min, max, default, callback, icon) 
MainTab:CreateSlider("WalkSpeed", 16, 250, 16, function(value)
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = value
end, "rbxassetid://7733798747") -- Flame Icon

Buttons

CreateButton (name, callback, icon) 
MainTab:CreateButton("Rejoin Server", function()
    -- Script Rejoin
end, "rbxassetid://7734051052") -- Refresh Icon

Dropdowns NEW

CreateDropdown (name, list, callback) 
MainTab:CreateDropdown("Teleport To", {"Shop", "Forest", "Dungeon"}, function(selected)
    print("Selected: ", selected)
end)

Keybinds NEW

CreateKeybind (name, default, callback) 
MainTab:CreateKeybind("Fly Key", Enum.KeyCode.F, function()
    print("Key Pressed!")
end)

Color Pickers NEW

CreateColorPicker (name, default, callback) 
MainTab:CreateColorPicker("UI Theme", Color3.fromRGB(255, 0, 0), function(color)
    print("Color changed to: ", color)
end)

Paragraphs

CreateParagraph (title, content, icon) 
MainTab:CreateParagraph("Credits", "Lib created by Ghost Team. Icons by Lucide.", "7733964719")
